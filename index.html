Node.js示例代码


// ChatGPT API调用示例 (Node.js)
const axios = require('axios');

const API_KEY = 'your_4z_api_key'; // 替换为您的4Z API密钥
const API_URL = 'https://api.zzzzapi.com/v1/chat/completions';

const headers = {
  'Authorization': `Bearer ${API_KEY}`,
  'Content-Type': 'application/json'
};

const data = {
  model: 'gpt-4o',
  messages: [
    {role: 'system', content: '你是一个专业的AI助手'},
    {role: 'user', content: '请为我的电商网站写一个产品描述'}
  ],
  temperature: 0.7,
  max_tokens: 2000
};

axios.post(API_URL, data, {headers})
  .then(response => {
    console.log(response.data.choices[0].message.content);
  })
  .catch(error => {
    console.error('调用出错:', error);
  });
